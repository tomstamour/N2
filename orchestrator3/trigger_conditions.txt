# trigger_conditions.txt — trade_mole launch rules for Orchestrator3.5.1
#
# Lines starting with '#' and blank lines are ignored.
# A '[group_name]' header starts a new rule group.
# Each condition line: <field> <op> <value>
#
# Within a group: all conditions are ANDed.
# Across groups:  the FIRST group whose conditions all pass fires (OR).
# If no group passes, no trade_mole is launched.
#
# Fields (case-sensitive):
#   Numeric: Float, LastDailyClosePrice, positive, negative, neutral,
#            sentiment_score, nocoref_neutral_filter,
#            nocoref_confidence_weighted, nocoref_net_score, nocoref_top_k,
#            nocoref_positional
#   String:  label, Exchange, Symbol  (case-insensitive comparison)
#
# Operators: <, <=, >, >=, =, !=
#   Numeric fields: all six operators allowed.
#   String  fields: only = and != allowed.
#   For = / != the parser tries float() first, then falls back to string.
#
# None on the left (lookup missed / pipeline failed / timeout) ALWAYS fails
# that condition — so a group cannot fire on missing data.
#
# Headline excluded_strings (managed via excluded_strings-2.txt) are an
# implicit gate ANDed with every group — a hit there blocks ALL launches.
#
# ─── Active rules ────────────────────────────────────────────────────────────

# Backwards-compatible default — matches Orchestrator3.5 thresholds.
[current_default]
sentiment_score > 0.7
Float < 50
